### 8.4.4 消息封装（Message.java）

```java
package com.textbook.chapter8;

import com.fasterxml.jackson.databind.ObjectMapper;

import java.util.Map;

/**
 * 消息封装：Agent间通信的数据结构
 * @since v0.2.0
 */
public class Message {
    private static final ObjectMapper objectMapper = new ObjectMapper();

    private String id;                    // 消息ID
    private String type;                  // 消息类型（task/response/error）
    private String source;                // 消息来源
    private String target;                // 消息目标
    private Map<String, Object> payload;  // 消息内容
    private long timestamp;               // 时间戳

    // --- 构造函数 ---
    public Message(String id, String type, String source, String target, Map<String, Object> payload) {
        this.id = id;
        this.type = type;
        this.source = source;
        this.target = target;
        this.payload = payload;
        this.timestamp = System.currentTimeMillis();
    }

    // --- Getter / Setter ---
    public String getId() { return id; }
    public String getType() { return type; }
    public String getSource() { return source; }
    public String getTarget() { return target; }
    public Map<String, Object> getPayload() { return payload; }
    public long getTimestamp() { return timestamp; }

    // --- JSON序列化 ---
    public String toJson() {
        try {
            return objectMapper.writeValueAsString(this);
        } catch (Exception e) {
            throw new RuntimeException("JSON序列化失败", e);
        }
    }

    // --- JSON反序列化 ---
    public static Message fromJson(String json) {
        try {
            return objectMapper.readValue(json, Message.class);
        } catch (Exception e) {
            throw new RuntimeException("JSON反序列化失败", e);
        }
    }
}
```

### 8.4.5 消息队列客户端（MessageQueueClient.java）

```java
package com.textbook.chapter8;

import org.apache.kafka.clients.consumer.KafkaConsumer;
import org.apache.kafka.clients.producer.KafkaProducer;
import org.apache.kafka.clients.producer.ProducerRecord;

import java.util.Properties;

/**
 * 消息队列客户端（Kafka实现）
 * @since v3.6.0
 */
public class MessageQueueClient {
    private final KafkaProducer<String, String> producer;
    private final KafkaConsumer<String, String> consumer;

    public MessageQueueClient() {
        // 1. 初始化Producer
        Properties producerProps = new Properties();
        producerProps.put("bootstrap.servers", ConfigLoader.get("KAFKA_BOOTSTRAP_SERVERS", "localhost:9092"));
        producerProps.put("key.serializer", "org.apache.kafka.common.serialization.StringSerializer");
        producerProps.put("value.serializer", "org.apache.kafka.common.serialization.StringSerializer");
        this.producer = new KafkaProducer<>(producerProps);

        // 2. 初始化Consumer（简化版，实际需配置group.id）
        Properties consumerProps = new Properties();
        consumerProps.put("bootstrap.servers", ConfigLoader.get("KAFKA_BOOTSTRAP_SERVERS", "localhost:9092"));
        consumerProps.put("key.deserializer", "org.apache.kafka.common.serialization.StringDeserializer");
        consumerProps.put("value.deserializer", "org.apache.kafka.common.serialization.StringDeserializer");
        this.consumer = new KafkaConsumer<>(consumerProps);
    }

    /**
     * 发送消息
     * @param topic Topic名称
     * @param message 消息内容

    public void publish(String topic, String message) {
```

### 8.4.7 轮询调度（RoundRobinStrategy.java）

```java
package com.textbook.chapter8;

import java.util.List;
import java.util.concurrent.atomic.AtomicInteger;

/**
 * 轮询调度策略
 * @since v0.2.0
 */
public class RoundRobinStrategy implements SchedulingStrategy {
    private final AtomicInteger counter = new AtomicInteger(0);

    @Override
    public String selectWorker(List<String> workers, Message message) {
        if (workers.isEmpty()) {
            throw new IllegalArgumentException("Worker列表为空");
        }
        
        int index = counter.getAndIncrement() % workers.size();
        return workers.get(index);
    }
}
```

```java
package com.textbook.chapter8;

import java.util.List;

/**
 * 随机调度策略

 */
public class RandomStrategy implements SchedulingStrategy {
    
    @Override
    public String selectWorker(List<String> workers, Message message) {
```

### 8.4.10 Manager Agent（ManagerAgent.java）

```java
package com.textbook.chapter8;

import java.util.ArrayList;
import java.util.List;
import java.util.Map;

/**
 * Manager Agent：接收用户请求、分配任务、整合结果
 * @since v0.2.0
 */
public class ManagerAgent {
    private final String name;
    private final MessageQueueClient messageQueueClient;
    private final SchedulingStrategy schedulingStrategy;
    private final List<String> workers;
    private final List<Message> pendingTasks = new ArrayList<>();

    public ManagerAgent(String name, MessageQueueClient messageQueueClient,
                       SchedulingStrategy schedulingStrategy, List<String> workers) {
        this.name = name;
        this.messageQueueClient = messageQueueClient;
        this.schedulingStrategy = schedulingStrategy;
        this.workers = workers;
    }

    /**
     * 接收用户请求
     * @param request 用户请求内容
     */
    public void handleRequest(String request) {
        // 1. 创建任务消息
        Message task = new Message(
            "task_" + System.currentTimeMillis(),
            "task",
            name,
            schedulingStrategy.selectWorker(workers, null),
            Map.of("request", request)
        );

        // 2. 发送到消息队列
        messageQueueClient.publish("tasks", task.toJson());

        // 3. 记录待处理任务
        pendingTasks.add(task);
    }

    /**
     * 整合Worker结果
     * @param results Worker结果列表
     */
    public void integrateResults(List<Message> results) {
        // TODO: 实际整合逻辑
        // 简化版：打印结果
        results.forEach(result -> {
            System.out.println("整合结果：" + result.getPayload());
        });
    }

    /**
     * 清空待处理任务
     */
    public void clearPendingTasks() {
        pendingTasks.clear();
    }
}
```

### 8.4.13 主系统类（MultiAgentSystem.java）

```java
package com.textbook.chapter8;

import java.util.Arrays;
import java.util.List;

/**
 * 多Agent系统主类
 * @since v0.2.0
 */
public class MultiAgentSystem {
    private final ManagerAgent managerAgent;
    private final List<WorkerAgent> workerAgents;
    private final ObserverAgent observerAgent;
    private final MessageQueueClient messageQueueClient;

    public MultiAgentSystem() {
        // 1. 初始化消息队列客户端
        this.messageQueueClient = new MessageQueueClient();

        // 2. 初始化调度策略
        SchedulingStrategy strategy = new RoundRobinStrategy();

        // 3. 初始化Worker列表
        List<String> workerNames = Arrays.asList("ImageWorker", "DataWorker", "TextWorker");

        // 4. 初始化Manager Agent
        this.managerAgent = new ManagerAgent("ManagerAgent", messageQueueClient, strategy, workerNames);

        // 5. 初始化Worker Agents
        this.workerAgents = workerNames.stream()
            .map(WorkerAgent::new)
            .toList();

        // 6. 初始化Observer Agent
        this.observerAgent = new ObserverAgent("ObserverAgent");
    }

    /**
     * 处理用户请求
     * @param request 用户请求内容
     */
    public void processRequest(String request) {
        // 1. Manager接收请求
        managerAgent.handleRequest(request);

        // 2. Observer监控
        observerAgent.monitor();

        // 3. 打印任务分配信息
        System.out.println("任务已分配到Worker");
    }

    /**
     * 关闭系统
     */
    public void shutdown() {
        messageQueueClient.close();
        System.out.println("多Agent系统已关闭");
    }
}
```

### 8.4.14 启动入口（Main.java）

```java
package com.textbook.chapter8;

public class Main {
    public static void main(String[] args) {
        // 1. 初始化多Agent系统
        MultiAgentSystem system = new MultiAgentSystem();

        // 2. 处理用户请求（简化版）
        System.out.println("=== 多Agent协作系统 ===");
        System.out.println("输入 'quit' 退出");

        java.util.Scanner scanner = new java.util.Scanner(System.in);
        
        while (true) {
            System.out.print("用户：");
            String input = scanner.nextLine();
            if ("quit".equalsIgnoreCase(input)) {
                break;
            }

            system.processRequest(input);
        }

        // 3. 关闭系统
        system.shutdown();
    }
}
```
