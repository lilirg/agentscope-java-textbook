**src/main/java/com/agentscope/arch/AgentArchitectureExample.java**：
```java
package com.agentscope.arch;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * Agent架构最小可运行示例
 * 
 * @since agentscope-java 0.2.0
 * @see <a href="https://github.com/lilirg/agentscope-java">agentscope-java</a>
 */
public class AgentArchitectureExample {
    private static final Logger logger = LoggerFactory.getLogger(AgentArchitectureExample.class);

    /**
     * Agent架构核心组件：Memory + Tool + Planning + Execution
     * 
     * @param input 用户输入
     * @return Agent处理结果
     */
    public String process(String input) {
        logger.info("Agent Architecture processing input: {}", input);

        // Memory：存储历史信息
        String memory = "历史对话：无";

        // Tool：调用外部API
        String tool = "调用天气API";

        // Planning：制定行动方案
        String planning = "获取天气信息";

        // Execution：执行具体操作
        String execution = "返回天气信息";

        return execution;
    }

    /**
     * 主方法，运行Agent架构最小示例
     */
    public static void main(String[] args) {
        logger.info("Agent Architecture Example");
        
        AgentArchitectureExample example = new AgentArchitectureExample();
        String result = example.process("你好，请问今天天气如何？");
        
        logger.info("Agent Architecture result: {}", result);
    }
}
```

## 4.4.5 Memory系统最小可运行示例

**src/main/java/com/agentscope/arch/MemorySystemExample.java**：
```java
package com.agentscope.arch;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.HashMap;
import java.util.Map;

/**
 * Memory系统最小可运行示例
 * 
 * @since agentscope-java 0.2.0
 * @see <a href="https://github.com/lilirg/agentscope-java">agentscope-java</a>
 */
public class MemorySystemExample {
    private static final Logger logger = LoggerFactory.getLogger(MemorySystemExample.class);

    // 短期记忆：存储当前对话历史
    private final Map<String, String> shortTermMemory = new HashMap<>();

    // 长期记忆：存储长期知识库
    private final Map<String, String> longTermMemory = new HashMap<>();

    // 上下文记忆：存储任务上下文
    private final Map<String, String> contextMemory = new HashMap<>();

    /**
     * 存储短期记忆
     * 
     * @param key 键
     * @param value 值
     */
    public void storeShortTermMemory(String key, String value) {
        logger.info("Storing short-term memory: {} -> {}", key, value);
        shortTermMemory.put(key, value);
    }

    /**
     * 存储长期记忆
     * 
     * @param key 键
     * @param value 值
     */
    public void storeLongTermMemory(String key, String value) {
        logger.info("Storing long-term memory: {} -> {}", key, value);
        longTermMemory.put(key, value);
    }

    /**
     * 存储上下文记忆
     * 
     * @param key 键
     * @param value 值
     */
    public void storeContextMemory(String key, String value) {
        logger.info("Storing context memory: {} -> {}", key, value);
        contextMemory.put(key, value);
    }

    /**
     * 获取短期记忆
     * 
     * @param key 键
     * @return 值
     */
    public String getShortTermMemory(String key) {
        logger.info("Getting short-term memory: {}", key);
        return shortTermMemory.get(key);
    }

    /**
     * 获取长期记忆
     * 
     * @param key 键
     * @return 值
     */
    public String getLongTermMemory(String key) {
        logger.info("Getting long-term memory: {}", key);
        return longTermMemory.get(key);
    }

    /**
     * 获取上下文记忆
     * 
     * @param key 键
     * @return 值
     */
    public String getContextMemory(String key) {
        logger.info("Getting context memory: {}", key);
        return contextMemory.get(key);
    }

    /**
     * 主方法，运行Memory系统最小示例
     */
    public static void main(String[] args) {
        logger.info("Memory System Example");
        
        MemorySystemExample example = new MemorySystemExample();
        
        example.storeShortTermMemory("current_question", "今天天气如何？");
        example.storeLongTermMemory("user_location", "北京");
        example.storeContextMemory("task", "天气查询");
        
        logger.info("Short-term memory: {}", example.getShortTermMemory("current_question"));
        logger.info("Long-term memory: {}", example.getLongTermMemory("user_location"));
        logger.info("Context memory: {}", example.getContextMemory("task"));
    }
}
```

## 4.4.6 Tool系统最小可运行示例

**src/main/java/com/agentscope/arch/ToolSystemExample.java**：
```java
package com.agentscope.arch;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.HashMap;
import java.util.Map;

/**
 * Tool系统最小可运行示例
 * 
 * @since agentscope-java 0.2.0
 * @see <a href="https://github.com/lilirg/agentscope-java">agentscope-java</a>
 */
public class ToolSystemExample {
    private static final Logger logger = LoggerFactory.getLogger(ToolSystemExample.class);

    // 工具注册表
    private final Map<String, Tool> toolRegistry = new HashMap<>();

    /**
     * 工具接口
     */
    public interface Tool {
        String execute(String input);
    }

    /**
     * 注册工具
     * 
     * @param name 工具名称
     * @param tool 工具
     */
    public void registerTool(String name, Tool tool) {
        logger.info("Registering tool: {}", name);
        toolRegistry.put(name, tool);
    }

    /**
     * 调用工具
     * 
     * @param name 工具名称
     * @param input 输入
     * @return 工具执行结果
     */
    public String callTool(String name, String input) {
        logger.info("Calling tool: {}", name);

        Tool tool = toolRegistry.get(name);
        if (tool == null) {
            logger.error("Tool not found: {}", name);
            return "Error: Tool not found";
        }

        return tool.execute(input);
    }

    /**
     * 主方法，运行Tool系统最小示例
     */
    public static void main(String[] args) {
        logger.info("Tool System Example");
        
        ToolSystemExample example = new ToolSystemExample();
        
        // 注册天气工具
        example.registerTool("weather", input -> "天气信息：晴朗，气温25度");
        
        // 注册搜索工具
        example.registerTool("search", input -> "搜索结果：相关信息");
        
        // 调用天气工具
        String weatherResult = example.callTool("weather", "北京");
        logger.info("Weather result: {}", weatherResult);
        
        // 调用搜索工具
        String searchResult = example.callTool("search", "AI Agent");
        logger.info("Search result: {}", searchResult);
    }
}
```

## 4.4.7 Planning系统最小可运行示例

**src/main/java/com/agentscope/arch/PlanningSystemExample.java**：
```java
package com.agentscope.arch;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.ArrayList;
import java.util.List;

/**
 * Planning系统最小可运行示例
 * 
 * @since agentscope-java 0.2.0
 * @see <a href="https://github.com/lilirg/agentscope-java">agentscope-java</a>
 */
public class PlanningSystemExample {
    private static final Logger logger = LoggerFactory.getLogger(PlanningSystemExample.class);

    /**
     * 目标
     */
    public static class Goal {
        private final String description;
        private final List<SubGoal> subGoals = new ArrayList<>();

        public Goal(String description) {
            this.description = description;
        }

        public void addSubGoal(SubGoal subGoal) {
            subGoals.add(subGoal);
        }

        public List<SubGoal> getSubGoals() {
            return subGoals;
        }

        public String getDescription() {
            return description;
        }
    }

    /**
     * 子目标
     */
    public static class SubGoal {
        private final String description;
        private final String action;

        public SubGoal(String description, String action) {
            this.description = description;
            this.action = action;
        }

        public String getDescription() {
            return description;
        }

        public String getAction() {
            return action;
        }
    }

    /**
     * 目标分解
     * 
     * @param goal 目标
     * @return 子目标列表
     */
    public List<SubGoal> decomposeGoal(String goal) {
        logger.info("Decomposing goal: {}", goal);

        List<SubGoal> subGoals = new ArrayList<>();
        
        // 示例：天气查询分解
        if (goal.contains("天气")) {
            subGoals.add(new SubGoal("获取用户位置", "获取用户位置"));
            subGoals.add(new SubGoal("查询天气API", "查询天气API"));
            subGoals.add(new SubGoal("返回天气信息", "返回天气信息"));
        }
        
        return subGoals;
    }

    /**
     * 主方法，运行Planning系统最小示例
     */
    public static void main(String[] args) {
        logger.info("Planning System Example");
        
        PlanningSystemExample example = new PlanningSystemExample();
        
        Goal goal = new Goal("查询北京天气");
        List<SubGoal> subGoals = example.decomposeGoal("查询北京天气");
        
        logger.info("Goal: {}", goal.getDescription());
        for (SubGoal subGoal : subGoals) {
            logger.info("SubGoal: {} -> Action: {}", subGoal.getDescription(), subGoal.getAction());
        }
    }
}
```

## 4.4.8 Execution系统最小可运行示例

**src/main/java/com/agentscope/arch/ExecutionSystemExample.java**：
```java
package com.agentscope.arch;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.ArrayList;
import java.util.List;

/**
 * Execution系统最小可运行示例
 * 
 * @since agentscope-java 0.2.0
 * @see <a href="https://github.com/lilirg/agentscope-java">agentscope-java</a>
 */
public class ExecutionSystemExample {
    private static final Logger logger = LoggerFactory.getLogger(ExecutionSystemExample.class);

    /**
     * 任务执行
     * 
     * @param task 任务
     * @return 任务执行结果
     */
    public String executeTask(String task) {
        logger.info("Executing task: {}", task);

        try {
            // 执行任务
            String result = "任务执行成功: " + task;
            
            // 反馈结果
            logger.info("Task execution result: {}", result);
            
            return result;
        } catch (Exception e) {
            logger.error("Task execution failed", e);
            
            // 错误处理
            return "任务执行失败: " + e.getMessage();
        }
    }

    /**
     * 主方法，运行Execution系统最小示例
     */
    public static void main(String[] args) {
        logger.info("Execution System Example");
        
        ExecutionSystemExample example = new ExecutionSystemExample();
        
        String result = example.executeTask("查询北京天气");
        logger.info("Execution result: {}", result);
        
        try {
            example.executeTask("throw error");
        } catch (Exception e) {
            logger.error("Execution failed", e);
        }
    }
}
```
