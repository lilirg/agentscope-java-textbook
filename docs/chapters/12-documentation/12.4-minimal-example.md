#### 6. Agent核心代码

```java
package com.textbook.chapter12;

/**
 * Agent服务类
 * 提供Agent查询、处理、监控等功能
 * 
 * @author Lilirg
 * @since v12.4.0
 */
@Service
public class AgentService {
    
    private final AgentClient agentClient;
    private final CacheManager cacheManager;
    private final PerformanceMonitor performanceMonitor;
    
    public AgentService(AgentClient agentClient, CacheManager cacheManager, PerformanceMonitor performanceMonitor) {
        this.agentClient = agentClient;
        this.cacheManager = cacheManager;
        this.performanceMonitor = performanceMonitor;
    }

    /**
     * 处理用户查询
     * 
     * @param query 用户查询内容
     * @return 处理结果
     */
    public String processQuery(String query) {
        // 1. 查缓存
        String cached = cacheManager.get(query);
        if (cached != null) {
            return cached;
        }

        // 2. 调用Agent处理
        long startTime = System.currentTimeMillis();
        String result = agentClient.call(query);
        long responseTime = System.currentTimeMillis() - startTime;

        // 3. 监控
        performanceMonitor.monitorResponseTime(responseTime);

        // 4. 写缓存
        cacheManager.put(query, result);

        return result;
    }
}
```
