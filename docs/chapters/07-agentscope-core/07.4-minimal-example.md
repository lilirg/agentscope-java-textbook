### 7.4.3 自定义工具（CustomTool.java）

```java
package com.textbook.chapter7;

import com.alibaba.agentscope.core.tool.Tool;

import java.util.Map;

/**
 * 自定义工具：演示AgentScope-Java的Tool接口实现
 * @since v0.2.0
 */
public class CustomTool implements Tool {
    private final String name;
    private final String description;

    public CustomTool(String name, String description) {
        this.name = name;
        this.description = description;
    }

    /**
     * 执行工具逻辑
     * @param parameters 工具参数
     * @return 工具输出
     */
    @Override
    public String execute(Map<String, Object> parameters) {
        System.out.println("执行工具：" + name);
        System.out.println("参数：" + parameters);
        
        // 示例逻辑：返回当前时间
        return "当前时间：" + java.time.LocalDateTime.now();
    }

    /**
     * 获取工具名称
     * @return 工具名称
     */
    @Override
    public String getName() {
        return name;
    }

    /**
     * 获取工具描述
     * @return 工具描述
     */
    @Override
    public String getDescription() {
        return description;
    }
}
```

### 7.4.4 自定义Agent（CustomAgent.java）

```java
package com.textbook.chapter7;

import com.alibaba.agentscope.core.agent.Agent;
import com.alibaba.agentscope.core.response.Response;

import java.util.List;

/**
 * 自定义Agent：演示AgentScope-Java的Agent继承
 * @since v0.2.0
 */
public class CustomAgent extends Agent {
    
    public CustomAgent(String name) {
        super(name);
    }

    /**
     * 实现run方法
     * @param input 用户输入
     * @return Agent响应
     */
    @Override
    public Response run(String input) {
        System.out.println("收到用户输入：" + input);
        
        // 构建提示词
        StringBuilder prompt = new StringBuilder();
        prompt.append("你是").append(getName()).append("，一个AI助手。\n");
        prompt.append("用户输入：").append(input).append("\n");
        
        // 添加工具描述（用于提示词）
        List<Tool> tools = getTools();
        if (!tools.isEmpty()) {
            prompt.append("\n可用工具：\n");
            for (Tool tool : tools) {
                prompt.append("- ").append(tool.getName()).append("：")
                      .append(tool.getDescription()).append("\n");
            }
        }
        
        // TODO: 实际调用LLM API（实际使用时需集成OpenAI/通义千问API）
        // 这里使用模拟响应
        String responseContent = "已收到输入：" + input + "（实际使用时需集成LLM API）";
        
        return new Response(responseContent);
    }

    /**
     * 获取工具列表（受保护方法）
     * @return 工具列表
     */
    protected List<Tool> getTools() {
        return tools;
    }
}
```

### 7.4.5 Agent服务（AgentService.java）

```java
package com.textbook.chapter7;

import org.springframework.stereotype.Service;

/**
 * Agent服务：演示Agent与Spring Boot集成
 * @since v0.2.0
 */
@Service
public class AgentService {
    private final CustomAgent agent;

    public AgentService() {
        // 创建Agent
        agent = new CustomAgent("CustomAgent");
        
        // 添加自定义工具
        agent.addTool(new CustomTool("currentTime", "获取当前时间"));
    }

    /**
     * 执行Agent
     * @param input 用户输入
     * @return Agent响应
     */
    public String execute(String input) {
        return agent.run(input).getContent();
    }
}
```

import org.junit.jupiter.api.Test;
