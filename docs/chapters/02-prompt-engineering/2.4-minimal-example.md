## 提示词工程最小可运行示例

**src/main/java/com/agentscope/prompt/PromptExample.java**：
```java
package com.agentscope.prompt;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import com.openai.OpenAI;
import com.openai.models.ChatCompletionRequest;
import com.openai.models.Message;
import com.openai.models.Role;
import com.openai.models.Choice;

import java.util.List;

/**
 * 提示词工程最小可运行示例
 * 
 * @since OpenAI SDK 0.28.0
 * @see <a href="https://github.com/openai/openai-java">openai-java</a>
 */
public class PromptExample {
    private static final Logger logger = LoggerFactory.getLogger(PromptExample.class);

    private final OpenAI openAI;

    public PromptExample(String apiKey) {
        this.openAI = OpenAI.builder()
            .apiKey(apiKey)
            .build();
    }

    /**
     * 提示词组成部分：指令 + 上下文 + 示例
     * 
     * @param instruction 指令
     * @param context 上下文
     * @param examples 示例
     * @return 完整提示词
     */
    public String buildPrompt(String instruction, String context, String examples) {
        logger.info("Building prompt");

        // 构建提示词
        String prompt = instruction + "\n\n" + context + "\n\n" + examples;
        
        return prompt;
    }

    /**
     * 调用LLM生成结果
     * 
     * @param prompt 提示词
     * @return LLM生成结果
     */
    public String generate(String prompt) {
        logger.info("Generating result with prompt: {}", prompt);

        try {
            // 创建ChatCompletionRequest
            ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("gpt-3.5-turbo")
                .messages(List.of(
                    Message.builder()
                        .role(Role.USER)
                        .content(prompt)
                        .build()
                ))
                .build();

            // 调用LLM
            List<Choice> choices = openAI.createChatCompletion(request).getChoices();
            
            return choices.get(0).getMessage().getContent();
        } catch (Exception e) {
            logger.error("Failed to generate result", e);
            return "Error: " + e.getMessage();
        }
    }

    /**
     * 主方法，运行提示词工程最小示例
     */
    public static void main(String[] args) {
        logger.info("Prompt Engineering Example");
        
        // 从环境变量读取API Key（生产环境）
        // 或使用dotenv-java加载.env文件（开发环境）
        String apiKey = System.getenv("OPENAI_API_KEY");
        if (apiKey == null || apiKey.isEmpty()) {
            logger.warn("OPENAI_API_KEY not found in environment variables");
        }
        
        PromptExample example = new PromptExample(apiKey);
        
        String prompt = example.buildPrompt(
            "你是一个天气助手，请回答天气相关问题。",
            "当前日期：2024-04-15\n用户位置：北京",
            "Q: 今天天气如何？\nA: 今天晴朗，气温25度。"
        );
        
        logger.info("Prompt: {}", prompt);
        
        String result = example.generate(prompt);
        
        logger.info("Result: {}", result);
    }
}
```

**src/main/java/com/agentscope/prompt/PromptOptimizationExample.java**：
```java
package com.agentscope.prompt;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import com.openai.OpenAI;
import com.openai.models.ChatCompletionRequest;
import com.openai.models.Message;
import com.openai.models.Role;

/**
 * 提示词优化最小可运行示例
 * 
 * @since OpenAI SDK 0.28.0
 * @see <a href="https://github.com/openai/openai-java">openai-java</a>
 */
public class PromptOptimizationExample {
    private static final Logger logger = LoggerFactory.getLogger(PromptOptimizationExample.class);

    private final OpenAI openAI;

    public PromptOptimizationExample(String apiKey) {
        this.openAI = OpenAI.builder()
            .apiKey(apiKey)
            .build();

     * 优化提示词：增加约束条件
     * 
     * @param prompt 原始提示词
     * @return 优化后的提示词
     */
    public String optimizePrompt(String prompt) {
        logger.info("Optimizing prompt");

        // 增加约束条件
        String optimizedPrompt = prompt + "\n\n约束条件：\n1. 回答不超过100字\n2. 使用中文\n3. 包含具体数据";
        
        return optimizedPrompt;
    }

    /**
     * 调用LLM生成结果
     * 
     * @param prompt 提示词
     * @return LLM生成结果
     */
    public String generate(String prompt) {
        logger.info("Generating result with prompt: {}", prompt);

        try {
            // 创建ChatCompletionRequest
            ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("gpt-3.5-turbo")
                .messages(List.of(
                    Message.builder()
                        .role(Role.USER)
                        .content(prompt)
                        .build()
                ))
                .build();

            // 调用LLM
            List<Choice> choices = openAI.createChatCompletion(request).getChoices();
            
            return choices.get(0).getMessage().getContent();
        } catch (Exception e) {
            logger.error("Failed to generate result", e);
            return "Error: " + e.getMessage();
        }
    }

    /**
     * 主方法，运行提示词优化最小示例
     */
    public static void main(String[] args) {
        logger.info("Prompt Optimization Example");
        
        // 从环境变量读取API Key（生产环境）
        String apiKey = System.getenv("OPENAI_API_KEY");
        if (apiKey == null || apiKey.isEmpty()) {
            logger.warn("OPENAI_API_KEY not found in environment variables");
        }
        
        PromptOptimizationExample example = new PromptOptimizationExample(apiKey);
        
        String prompt = "你是一个天气助手，请回答天气相关问题。";
        String optimizedPrompt = example.optimizePrompt(prompt);
        
        logger.info("Optimized Prompt: {}", optimizedPrompt);
        
        String result = example.generate(optimizedPrompt);
        
        logger.info("Result: {}", result);
    }
}
```

## 提示词评估最小可运行示例

**src/main/java/com/agentscope/prompt/PromptEvaluationExample.java**：
```java
package com.agentscope.prompt;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * 提示词评估最小可运行示例
 * 
 * @since OpenAI SDK 0.28.0
 * @see <a href="https://github.com/openai/openai-java">openai-java</a>
 */
public class PromptEvaluationExample {
    private static final Logger logger = LoggerFactory.getLogger(PromptEvaluationExample.class);

    /**
     * 评估提示词：准确性
     * 
     * @param expectedOutput 期望输出
     * @param actualOutput 实际输出
     * @return 准确性评分（0-100）
     */
    public int evaluateAccuracy(String expectedOutput, String actualOutput) {
        logger.info("Evaluating accuracy");

        // 简单字符串匹配（实际应用中可使用更复杂的算法）
        if (expectedOutput.equals(actualOutput)) {
            return 100;
        } else {
            return 0;
        }
    }

    /**
     * 评估提示词：完整性
     * 
     * @param expectedOutput 期望输出
     * @param actualOutput 实际输出
     * @return 完整性评分（0-100）
     */
    public int evaluateCompleteness(String expectedOutput, String actualOutput) {
        logger.info("Evaluating completeness");

        // 简单长度匹配（实际应用中可使用更复杂的算法）
        if (actualOutput.length() >= expectedOutput.length()) {
            return 100;
        } else {
            return (actualOutput.length() * 100) / expectedOutput.length();
        }
    }

    /**
     * 主方法，运行提示词评估最小示例
     */
    public static void main(String[] args) {
        logger.info("Prompt Evaluation Example");
        
        PromptEvaluationExample example = new PromptEvaluationExample();
        
        String expectedOutput = "今天晴朗，气温25度";
        String actualOutput = "今天晴朗，气温25度";
        
        int accuracy = example.evaluateAccuracy(expectedOutput, actualOutput);
        int completeness = example.evaluateCompleteness(expectedOutput, actualOutput);
        
        logger.info("Accuracy: {}", accuracy);
        logger.info("Completeness: {}", completeness);
    }
}
```
