### 1.4.4 Agent最小可运行示例

**src/main/java/com/agentscope/minimal/AgentMinimal.java**：
```java
package com.agentscope.minimal;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * Agent最小可运行示例
 * 
 * @since agentscope-java 0.2.0
 * @see <a href="https://github.com/lilirg/agentscope-java">agentscope-java</a>
 */
public class AgentMinimal {
    private static final Logger logger = LoggerFactory.getLogger(AgentMinimal.class);

    /**
     * Agent核心能力：感知、推理、决策、行动
     * 
     * @param input 用户输入
     * @return Agent处理结果
     */
    public String process(String input) {
        logger.info("Agent processing input: {}", input);

        // 感知：接收用户输入
        String perception = input;

        // 推理：基于知识库进行推理
        String reasoning = "基于输入：" + perception;

        // 决策：选择最优行动方案
        String decision = "执行决策";

        // 行动：执行具体操作
        String action = "返回结果";

        return action;
    }

    /**
     * 主方法，运行Agent最小示例
     */
    public static void main(String[] args) {
        logger.info("Agent Minimal Example");
        
        AgentMinimal agent = new AgentMinimal();
        String result = agent.process("你好，请问今天天气如何？");
        
        logger.info("Agent result: {}", result);
    }
}
```

### 1.4.5 提示词工程最小可运行示例

**src/main/java/com/agentscope/minimal/PromptExample.java**：
```java
package com.agentscope.minimal;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * 提示词工程最小可运行示例
 * 
 * @since agentscope-java 0.2.0
 * @see <a href="https://github.com/lilirg/agentscope-java">agentscope-java</a>
 */
public class PromptExample {
    private static final Logger logger = LoggerFactory.getLogger(PromptExample.class);

    /**
     * 提示词组成部分：指令 + 上下文 + 示例
     * 
     * @param instruction 指令
     * @param context 上下文
     * @param examples 示例
     * @return 完整提示词
     */
    public String buildPrompt(String instruction, String context, String examples) {
        logger.info("Building prompt");

        // 构建提示词
        String prompt = instruction + "\n\n" + context + "\n\n" + examples;
        
        return prompt;
    }

    /**
     * 主方法，运行提示词工程最小示例
     */
    public static void main(String[] args) {
        logger.info("Prompt Engineering Example");
        
        PromptExample example = new PromptExample();
        String prompt = example.buildPrompt(
            "你是一个天气助手，请回答天气相关问题。",
            "当前日期：2024-04-15\n用户位置：北京",
            "Q: 今天天气如何？\nA: 今天晴朗，气温25度。"
        );
        
        logger.info("Prompt: {}", prompt);
    }
}
```

### 1.4.6 API接入最小可运行示例

**src/main/java/com/agentscope/minimal/APIExample.java**：
```java
package com.agentscope.minimal;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.net.URI;

/**
 * API接入最小可运行示例
 * 
 * @since agentscope-java 0.2.0
 * @see <a href="https://github.com/lilirg/agentscope-java">agentscope-java</a>
 */
public class APIExample {
    private static final Logger logger = LoggerFactory.getLogger(APIExample.class);

    /**
     * RESTful API调用示例
     * 
     * @param url API URL
     * @return API响应
     */
    public String callAPI(String url) {
        logger.info("Calling API: {}", url);

        try {
            // 创建HTTP客户端
            HttpClient client = HttpClient.newHttpClient();

            // 创建HTTP请求
            HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(url))
                .GET()
                .build();

            // 发送请求并获取响应
            HttpResponse<String> response = client.send(request, HttpResponse.BodyHandlers.ofString());

            // 返回响应内容
            return response.body();
        } catch (Exception e) {
            logger.error("Failed to call API", e);
            return "Error: " + e.getMessage();
        }
    }

    /**
     * 主方法，运行API接入最小示例
     */
    public static void main(String[] args) {
        logger.info("API Example");

        APIExample example = new APIExample();
        String response = example.callAPI("https://jsonplaceholder.typicode.com/posts/1");
        
        logger.info("API Response: {}", response);
    }
}
```
