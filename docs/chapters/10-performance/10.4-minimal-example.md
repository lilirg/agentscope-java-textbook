```java
package com.textbook.chapter10;

/**
 * 提示词优化类
 * @since v10.4.0
 */
public class TokenOptimizer {
    
    public String optimize(String prompt) {
        // 1. 去除冗余描述
        String optimized = prompt.replaceAll("\\s+", " ");
        
        // 2. 使用占位符
        optimized = optimized.replaceAll("请回答以下问题：(.*)", "{query}");
        
        return optimized.trim();
    }
}
```

#### 3. API调用优化（BatchAPI.java + AsyncAgentService.java）

```java
package com.textbook.chapter10;

import com.google.common.collect.Lists;

import java.util.ArrayList;
import java.util.List;

/**
 * 批量API调用类
 * @since v10.4.0
 */
public class BatchAPI {
    private final APIClient apiClient;

    public BatchAPI(APIClient apiClient) {
        this.apiClient = apiClient;
    }

    public List<String> processBatchQueries(List<String> queries) {
        // 1. 分组（每组10个查询）
        List<List<String>> groups = Lists.partition(queries, 10);

        // 2. 批量处理
        List<String> results = new ArrayList<>();
        for (List<String> group : groups) {
            results.addAll(apiClient.batchProcess(group));
        }

        return results;
    }
}
```

```java
package com.textbook.chapter10;

import java.util.List;
import java.util.concurrent.CompletableFuture;
import java.util.stream.Collectors;

/**
 * 异步Agent服务类
 * @since v10.4.0
 */
public class AsyncAgentService {
    private final APIClient apiClient;

    public CompletableFuture<String> processQueryAsync(String query) {
        return CompletableFuture.supplyAsync(() -> {
            return apiClient.call(query);
        });
    }

    public CompletableFuture<List<String>> processBatchQueriesAsync(List<String> queries) {
        List<CompletableFuture<String>> futures = queries.stream()
            .map(query -> CompletableFuture.supplyAsync(() -> apiClient.call(query)))
            .collect(Collectors.toList());

        return CompletableFuture.allOf(futures.toArray(new CompletableFuture[0]))
            .thenApply(v -> futures.stream()
                .map(CompletableFuture::join)
                .collect(Collectors.toList()));
    }
}
```

#### 5. 性能监控（PerformanceMonitor.java）

```java
package com.textbook.chapter10;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * 性能监控类
 * @since v10.4.0
 */
public class PerformanceMonitor {
    private static final Logger logger = LoggerFactory.getLogger("performance");

    public void monitorResponseTime(long responseTime) {
        if (responseTime > 1000) { // 超过1秒
            logger.error("响应时间超限: {}ms", responseTime);
            sendAlert("响应时间超限", "响应时间：" + responseTime + "ms");
        }
    }

    public void monitorCPUUsage(double cpuUsage) {
        if (cpuUsage > 80) { // CPU使用率超过80%
            logger.error("CPU使用率超限: {}%", cpuUsage);
            sendAlert("CPU使用率超限", "CPU使用率：" + cpuUsage + "%");
        }
    }

    public void monitorMemoryUsage(double memoryUsage) {
        if (memoryUsage > 80) { // 内存使用率超过80%
            logger.error("内存使用率超限: {}%", memoryUsage);
            sendAlert("内存使用率超限", "内存使用率：" + memoryUsage + "%");
        }
    }

    private void sendAlert(String title, String message) {
        // 钉钉告警
        DingTalkNotifier.notify(title, message);
    }
}
```

#### 6. Agent服务（AgentService.java）

```java
package com.textbook.chapter10;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

/**
 * Agent服务类
 * @since v10.4.0
 */
@Service
public class AgentService {
    
    @Autowired
    private TokenOptimizer tokenOptimizer;
    
    @Autowired
    private CacheManager cacheManager;
    
    @Autowired
    private BatchAPI batchAPI;
    
    @Autowired
    private AsyncAgentService asyncAgentService;
    
    @Autowired
    private PerformanceMonitor performanceMonitor;

    public String processQuery(String query) {
        // 1. 优化提示词
        String optimizedPrompt = tokenOptimizer.optimize(query);

        // 2. 查缓存
        String cached = cacheManager.get(optimizedPrompt);
        if (cached != null) {
            return cached;
        }

        // 3. 调用Agent处理
        long startTime = System.currentTimeMillis();
        String result = apiClient.call(optimizedPrompt);
        long responseTime = System.currentTimeMillis() - startTime;

        // 4. 监控
        performanceMonitor.monitorResponseTime(responseTime);

        // 5. 写缓存
        cacheManager.put(optimizedPrompt, result);

        return result;
    }

    public List<String> processBatchQueries(List<String> queries) {
        return batchAPI.processBatchQueries(queries);
    }

    public CompletableFuture<String> processQueryAsync(String query) {
        return asyncAgentService.processQueryAsync(query);
    }
}
```

### 10.4.4 Docker与Kubernetes部署

#### 1. Docker部署

**Dockerfile**：
```dockerfile
# Dockerfile
FROM openjdk:17-jdk-slim

WORKDIR /app

COPY target/performance-system-1.0-SNAPSHOT.jar performance-system.jar

EXPOSE 8080

ENTRYPOINT ["java", "-jar", "performance-system.jar"]
```

**构建与运行**：
```bash
# 1. 构建镜像
docker build -t performance-system:latest .

# 2. 运行容器
docker run -p 8080:8080 performance-system:latest
```

#### 2. Kubernetes部署

**Deployment配置**：
```yaml
# deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: performance-system
spec:
  replicas: 3
  selector:
    matchLabels:
      app: performance-system
  template:
    metadata:
      labels:
        app: performance-system
    spec:
      containers:
      - name: performance-system
        image: performance-system:latest
        ports:
        - containerPort: 8080
        resources:
          requests:
            memory: "256Mi"
            cpu: "100m"
          limits:
            memory: "512Mi"
            cpu: "500m"
```

**Service配置**：
```yaml
# service.yaml
apiVersion: v1
kind: Service
metadata:
  name: performance-system-service
spec:
  selector:
    app: performance-system
  ports:
  - protocol: TCP
    port: 80
    targetPort: 8080
  type: LoadBalancer
```
