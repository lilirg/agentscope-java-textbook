## 图片预处理（ImagePreprocessor.java）

```java
package com.textbook.chapter6;

import org.opencv.core.Core;
import org.opencv.core.CvType;
import org.opencv.core.Mat;
import org.opencv.core.Size;
import org.opencv.imgcodecs.Imgcodecs;
import org.opencv.imgproc.Imgproc;

import java.util.Arrays;

/**
 * 图片预处理器：OpenCV解码、resize、归一化
 * @since v4.9.0
 */
public class ImagePreprocessor {
    static {
        System.loadLibrary(Core.NATIVE_LIBRARY_NAME);
    }

    /**
     * 解码图片字节数组
     * @param imageData 图片字节数组
     * @return Mat对象（BGR格式）
     */
    public static Mat decodeImage(byte[] imageData) {
        Mat mat = Imgcodecs.imdecode(new org.opencv.core.MatOfByte(imageData), Imgcodecs.IMREAD_COLOR);
        if (mat.empty()) {
            throw new IllegalArgumentException("图片解码失败");
        }
        return mat;
    }

    /**
     * resize图片到指定尺寸
     * @param image 原始图片（Mat）
     * @param width 宽度
     * @param height 高度
     * @return resize后的图片
     */
    public static Mat resizeImage(Mat image, int width, int height) {
        Mat resized = new Mat();
        Imgproc.resize(image, resized, new Size(width, height));
        return resized;
    }

    /**
     * 归一化图片到[0,1]
     * @param image 图片（Mat）
     * @return float数组（[C,H,W]格式）
     */
    public static float[] normalizeImage(Mat image) {
        // 转为float类型，归一化到[0,1]
        Mat normalized = new Mat();
        image.convertTo(normalized, CvType.CV_32F, 1.0/255.0);
        
        // 转为数组（[C,H,W]格式：通道优先）
        int channels = normalized.channels();
        int height = normalized.rows();
        int width = normalized.cols();
        int total = height * width * channels;
        
        float[] data = new float[total];
        normalized.get(0, 0, data);
        
        // 重排为[RGB, H, W]格式（OpenCV默认BGR）
        float[] rgbData = new float[total];
        for (int h = 0; h < height; h++) {
            for (int w = 0; w < width; w++) {
                for (int c = 0; c < channels; c++) {
                    int srcIndex = (h * width + w) * channels + c;
                    int dstIndex;
                    if (c == 0) { // B → R
                        dstIndex = (h * width + w) * channels + 2;
                    } else if (c == 2) { // R → B
                        dstIndex = (h * width + w) * channels + 0;
                    } else { // G unchanged
                        dstIndex = srcIndex;
                    }
                    rgbData[dstIndex] = data[srcIndex];
                }
            }
        }
        
        return rgbData;
    }

    /**
     * 完整预处理流程
     * @param imageData 图片字节数组
     * @param targetWidth 目标宽度（默认224）
     * @param targetHeight 目标高度（默认224）
     * @return 归一化后的float数组
     */
    public static float[] preprocess(byte[] imageData, int targetWidth, int targetHeight) {
        Mat mat = decodeImage(imageData);
        Mat resized = resizeImage(mat, targetWidth, targetHeight);
        return normalizeImage(resized);
    }

    /**
     * 默认预处理（224×224）
     */
    public static float[] preprocess(byte[] imageData) {
        return preprocess(imageData, 224, 224);
    }
}
```

## 语音预处理（SpeechPreprocessor.java）

```java
package com.textbook.chapter6;

import org.springframework.core.io.ClassPathResource;

import java.io.*;
import java.nio.file.Files;
import java.nio.file.Paths;

/**
 * 语音预处理器：音频格式转换、分段（Python wrapper）
 * @since v0.2.0
 */
public class SpeechPreprocessor {
    private static final String TEMP_WAV_PATH = "temp_input.wav";
    private static final String TEMP_SEGMENT_DIR = "temp_segments/";

    /**
     * 调用Python脚本进行音频格式转换
     * @param audioData 音频字节数组（mp3/wav/flac等）
     * @return 转换后的wav文件路径
     */
    public static String convertToWav(byte[] audioData) {
        try {
            // 1. 保存原始音频
            Files.write(Paths.get(TEMP_WAV_PATH), audioData);
            
            // 2. 调用Python脚本（假设已安装pydub）
            String pythonScript = "convert_audio.py";
            ProcessBuilder pb = new ProcessBuilder("python", pythonScript, TEMP_WAV_PATH);
            pb.inheritIO();
            Process process = pb.start();
            process.waitFor();
            
            // 3. 读取转换后的音频
            return TEMP_WAV_PATH;
        } catch (Exception e) {
            throw new RuntimeException("音频格式转换失败", e);
        }
    }

    /**
     * 音频分段（30秒窗口，重叠1秒）
     * @param audioPath 音频文件路径
     * @return 分段后的音频文件路径列表
     */
    public static String[] segmentAudio(String audioPath) {
        try {
            // 创建分段目录
            Files.createDirectories(Paths.get(TEMP_SEGMENT_DIR));
            
            // 调用Python脚本进行音频分段
            String pythonScript = "segment_audio.py";
            ProcessBuilder pb = new ProcessBuilder("python", pythonScript, audioPath, TEMP_SEGMENT_DIR);
            pb.inheritIO();
            Process process = pb.start();
            process.waitFor();
            
            // 读取分段文件列表
            File segmentDir = new File(TEMP_SEGMENT_DIR);
            File[] files = segmentDir.listFiles((dir, name) -> name.endsWith(".wav"));
            if (files == null) {
                return new String[0];
            }
            
            String[] paths = new String[files.length];
            for (int i = 0; i < files.length; i++) {
                paths[i] = files[i].getAbsolutePath();
            }
            
            return paths;
        } catch (Exception e) {
            throw new RuntimeException("音频分段失败", e);
        }
    }

    /**
     * 完整预处理流程
     * @param audioData 音频字节数组
     * @return 分段后的音频文件路径列表
     */
    public static String[] preprocess(byte[] audioData) {
        String wavPath = convertToWav(audioData);
        return segmentAudio(wavPath);
    }
}
```

**Python脚本：convert_audio.py**
```python
#!/usr/bin/env python3
# pip install pydub

from pydub import AudioSegment
import sys

def convert_audio(input_path, output_path):
    """音频格式转换（mp3/wav/flac → wav）"""
    try:
        audio = AudioSegment.from_file(input_path)
        audio = audio.set_channels(1)  # 单声道
        audio = audio.set_frame_rate(16000)  # 16kHz
        audio.export(output_path, format="wav")
        print(f"成功转换：{output_path}")
    except Exception as e:
        print(f"转换失败：{e}", file=sys.stderr)
        sys.exit(1)

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("用法: python convert_audio.py <input_audio_path>")
        sys.exit(1)
    
    input_path = sys.argv[1]
    output_path = "temp_input.wav"
    convert_audio(input_path, output_path)
```

**Python脚本：segment_audio.py**
```python
#!/usr/bin/env python3
# pip install pydub

from pydub import AudioSegment
import sys
import os

def segment_audio(audio_path, output_dir):
    """音频分段（30秒窗口，重叠1秒）"""
    try:
        audio = AudioSegment.from_file(audio_path)
        segment_length = 30  # 秒
        overlap = 1  # 秒
        
        os.makedirs(output_dir, exist_ok=True)
        segments = []
        
        for i, start in enumerate(range(0, len(audio), (segment_length - overlap) * 1000)):
            end = start + segment_length * 1000
            segment = audio[start:end]
            
            output_path = os.path.join(output_dir, f"segment_{i}.wav")
            segment.export(output_path, format="wav")
            segments.append(output_path)
            
            if end >= len(audio):
                break
                
        print(f"成功分段：{len(segments)}段")
        return segments
    except Exception as e:
        print(f"分段失败：{e}", file=sys.stderr)
        sys.exit(1)

if __name__ == "__main__":
    if len(sys.argv) != 3:
        print("用法: python segment_audio.py <input_audio_path> <output_dir>")
        sys.exit(1)
    
    input_path = sys.argv[1]
    output_dir = sys.argv[2]
    segment_audio(input_path, output_dir)
```

## 多模态输入封装（MultimodalInput.java）

```java
package com.textbook.chapter6;

import java.util.Arrays;

/**
 * 多模态输入封装
 * @since v0.2.0
 */
public class MultimodalInput {
    private final String text;  // 文本输入
    private final byte[] image; // 图片输入（字节数组）
    private final byte[] audio; // 语音输入（字节数组）

    private MultimodalInput(String text, byte[] image, byte[] audio) {
        this.text = text;
        this.image = image;
        this.audio = audio;
    }

    public static MultimodalInput ofText(String text) {
        return new MultimodalInput(text, null, null);
    }

    public static MultimodalInput ofImage(byte[] image) {
        return new MultimodalInput(null, image, null);
    }

    public static MultimodalInput ofAudio(byte[] audio) {
        return new MultimodalInput(null, null, audio);
    }

    public static MultimodalInput ofAll(String text, byte[] image, byte[] audio) {
        return new MultimodalInput(text, image, audio);
    }

    // --- Getter ---
    public String getText() {
        return text;
    }

    public byte[] getImage() {
        return image;
    }

    public byte[] getAudio() {
        return audio;
    }

    public boolean hasText() {
        return text != null && !text.isEmpty();
    }

    public boolean hasImage() {
        return image != null && image.length > 0;
    }

    public boolean hasAudio() {
        return audio != null && audio.length > 0;
    }

    @Override
    public String toString() {
        return "MultimodalInput{" +
                "text=" + (text != null ? text.substring(0, Math.min(20, text.length())) + "..." : "null") +
                ", image=" + (image != null ? image.length + " bytes" : "null") +
                ", audio=" + (audio != null ? audio.length + " bytes" : "null") +
                '}';
    }
}
```

## Agent主类（MultimodalAgent.java）

```java
package com.textbook.chapter6;

import com.alibaba.agentscope.core.agent.Agent;
import com.alibaba.agentscope.core.response.Response;

import java.util.ArrayList;
import java.util.List;

/**
 * 多模态Agent：支持图片/语音/文本输入
 * @since v0.2.0
 */
public class MultimodalAgent extends Agent {
    private final List<String> recentImages = new ArrayList<>();
    private final List<String> recentAudios = new ArrayList<>();

    public MultimodalAgent(String name) {
        super(name);
    }

    /**
     * 重写run方法：处理多模态输入
     * @param input 多模态输入
     * @return Agent响应
     */
    @Override
    public Response run(MultimodalInput input) {
        // 1. 预处理输入
        String context = buildContext(input);
        
        // 2. 调用Agent生成响应（使用OpenAI SDK调用LLM API）
        OpenAIClient openAIClient = new OpenAIClient(apiKey);
        ChatCompletionOptions options = new ChatCompletionOptions(Arrays.asList(
            new ChatMessage(ChatRole.USER, context)
        ));
        ChatCompletions response = openAIClient.getChatCompletions("gpt-3.5-turbo", options);
        String responseContent = response.getChoices().get(0).getMessage().getContent();
        Response agentResponse = new Response(responseContent, "success");

        // 3. 存储非文本输入（用于长期记忆）
        if (input.hasImage()) {
            recentImages.add("Image processed: " + input.getImage().length + " bytes");
        }
        if (input.hasAudio()) {
            recentAudios.add("Audio processed: " + input.getAudio().length + " bytes");
        }

        return agentResponse;
    }

    /**
     * 构建提示词上下文（多模态 + 文本）
     * @param input 多模态输入
     * @return 完整提示词
     */
    private String buildContext(MultimodalInput input) {
        StringBuilder context = new StringBuilder();

        // 1. 图片输入（OpenCV预处理）
        if (input.hasImage()) {
            context.append("【用户上传图片】\n");
            try {
                float[] features = ImagePreprocessor.preprocess(input.getImage());
                context.append("图片已预处理（224×224，归一化），特征向量维度：").append(features.length).append("\n");
            } catch (Exception e) {
                context.append("图片预处理失败：").append(e.getMessage()).append("\n");
            }
        }

        // 2. 语音输入（pydub预处理）
        if (input.hasAudio()) {
            context.append("【用户语音输入】\n");
            try {
                String[] segments = SpeechPreprocessor.preprocess(input.getAudio());
                context.append("语音已分段，共").append(segments.length).append("段\n");
            } catch (Exception e) {
                context.append("语音预处理失败：").append(e.getMessage()).append("\n");
            }
        }

        // 3. 文本输入
        if (input.hasText()) {
            context.append("【用户文字输入】\n").append(input.getText()).append("\n");
        }

        // 4. 最近处理的非文本输入（用于长期记忆）
        if (!recentImages.isEmpty() || !recentAudios.isEmpty()) {
            context.append("\n【历史非文本输入】\n");
            recentImages.forEach(img -> context.append("- ").append(img).append("\n"));
            recentAudios.forEach(audio -> context.append("- ").append(audio).append("\n"));
        }

        return context.toString();
    }

    /**
     * 清空最近处理记录
     */
    public void clearHistory() {
        recentImages.clear();
        recentAudios.clear();
    }
}
```

## 启动入口（Main.java）

```java
package com.textbook.chapter6;

public class Main {
    public static void main(String[] args) {
        // 1. 初始化多模态Agent
        MultimodalAgent agent = new MultimodalAgent("MultimodalAgent");

        // 2. 启动交互（简化版：循环读取输入）
        java.util.Scanner scanner = new java.util.Scanner(System.in);
        System.out.println("=== Multimodal Agent ===");
        System.out.println("输入 'quit' 退出");
        System.out.println("输入 'image <path>' 测试图片");
        System.out.println("输入 'audio <path>' 测试音频");

        while (true) {
            System.out.print("用户：");
            String inputLine = scanner.nextLine();
            if ("quit".equalsIgnoreCase(inputLine)) {
                break;
            }

            // 解析输入
            String[] parts = inputLine.split(" ", 2);
            String command = parts.length > 0 ? parts[0] : "";
            String arg = parts.length > 1 ? parts[1] : "";

            try {
                MultimodalInput input;
                
                if ("image".equalsIgnoreCase(command)) {
                    // 读取图片文件
                    byte[] imageData = Files.readAllBytes(Paths.get(arg));
                    input = MultimodalInput.ofImage(imageData);
                } else if ("audio".equalsIgnoreCase(command)) {
                    // 读取音频文件
                    byte[] audioData = Files.readAllBytes(Paths.get(arg));
                    input = MultimodalInput.ofAudio(audioData);
                } else {
                    // 默认文本输入
                    input = MultimodalInput.ofText(inputLine);
                }

                // 调用Agent
                // Response response = agent.run(input);
                // System.out.println("Agent：" + response.getContent());
                
                System.out.println("Agent：[待实现LLM调用，请参考第3章]"); // 临时占位

            } catch (Exception e) {
                System.err.println("处理失败：" + e.getMessage());

        }
    }
```
