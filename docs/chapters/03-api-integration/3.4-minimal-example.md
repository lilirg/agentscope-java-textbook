## 3.4.4 API基础最小可运行示例

**src/main/java/com/agentscope/api/APIExample.java**：
```java
package com.agentscope.api;

import org.apache.http.HttpResponse;
import org.apache.http.client.methods.HttpGet;
import org.apache.http.impl.client.CloseableHttpClient;
import org.apache.http.impl.client.HttpClients;
import org.apache.http.util.EntityUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.IOException;

/**
 * API基础最小可运行示例
 * 
 * @since HttpClient 4.5.14
 * @see <a href="https://hc.apache.org/httpcomponents-client-4.5.x/index.html">HttpClient</a>
 */
public class APIExample {
    private static final Logger logger = LoggerFactory.getLogger(APIExample.class);

    /**
     * 调用API
     * 
     * @param url API URL
     * @return API响应
     * @throws IOException API调用异常
     */
    public String callAPI(String url) throws IOException {
        logger.info("Calling API: {}", url);

        // 创建HttpClient
        try (CloseableHttpClient httpClient = HttpClients.createDefault()) {
            // 创建HTTP请求
            HttpGet request = new HttpGet(url);

            // 执行请求
            HttpResponse response = httpClient.execute(request);

            // 返回响应内容
            return EntityUtils.toString(response.getEntity());
        }
    }

    /**
     * 主方法，运行API基础最小示例
     */
    public static void main(String[] args) {
        logger.info("API Example");

        try {
            APIExample example = new APIExample();
            String response = example.callAPI("https://jsonplaceholder.typicode.com/posts/1");
            logger.info("API Response: {}", response);
        } catch (IOException e) {
            logger.error("Failed to call API", e);
        }
    }
}
```

```java
package com.agentscope.api;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.http.HttpMethod;
import org.springframework.http.ResponseEntity;
import org.springframework.web.client.RestTemplate;

import java.net.URI;
import java.net.URISyntaxException;

/**
 * RESTful API最小可运行示例
 * 
 * @since Spring Boot 3.2.5
 * @see <a href="https://spring.io/projects/spring-boot">Spring Boot</a>
 */
public class RESTfulAPIExample {
    private static final Logger logger = LoggerFactory.getLogger(RESTfulAPIExample.class);

    private final RestTemplate restTemplate = new RestTemplate();

    /**
     * 调用RESTful API（GET）
     * 
     * @param url API URL
     * @return API响应
     */
    public String callGetAPI(String url) {
        logger.info("Calling GET API: {}", url);

        try {
            ResponseEntity<String> response = restTemplate.getForEntity(url, String.class);
            return response.getBody();
        } catch (Exception e) {
            logger.error("Failed to call GET API", e);
            return "Error: " + e.getMessage();
        }
    }

    /**
     * 主方法，运行RESTful API最小示例
     */
    public static void main(String[] args) {
        logger.info("RESTful API Example");

        try {
            RESTfulAPIExample example = new RESTfulAPIExample();
            String response = example.callGetAPI("https://jsonplaceholder.typicode.com/posts/1");
            logger.info("API Response: {}", response);
        } catch (Exception e) {
            logger.error("Failed to call API", e);
        }
    }
}
```

## 3.4.6 API文档最小可运行示例

**src/main/java/com/agentscope/api/APIDocExample.java**：
```java
package com.agentscope.api;

import io.swagger.v3.oas.annotations.Operation;
import io.swagger.v3.oas.annotations.tags.Tag;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

/**
 * API文档最小可运行示例
 * 
 * @since Spring Boot 3.2.5
 * @see <a href="https://spring.io/projects/spring-boot">Spring Boot</a>
 */
@RestController
@RequestMapping("/api")
@Tag(name = "API", description = "API文档示例")
public class APIDocExample {
    private static final Logger logger = LoggerFactory.getLogger(APIDocExample.class);

    /**
     * 查询API
     */
    @GetMapping("/query")
    @Operation(summary = "查询API", description = "查询API数据")
    public ResponseEntity<String> queryAPI(@RequestParam String url) {
        logger.info("Querying API: {}", url);

        try {
            // 使用HttpClient调用API
            HttpClient client = HttpClient.newHttpClient();
            HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(url))
                .GET()
                .build();
            
            HttpResponse<String> response = client.send(request, HttpResponse.BodyHandlers.ofString());
            return ResponseEntity.ok(response.body());
        } catch (Exception e) {
            logger.error("Failed to query API", e);
            return ResponseEntity.internalServerError().body("Error: " + e.getMessage());
        }
    }

    /**
     * 主方法，运行API文档最小示例
     */
    public static void main(String[] args) {
        logger.info("API Doc Example");
    }
}
```

## 3.4.7 API安全最小可运行示例

**src/main/java/com/agentscope/api/APISecurityExample.java**：
```java
package com.agentscope.api;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.http.HttpHeaders;
import org.springframework.http.HttpRequest;
import org.springframework.http.client.ClientHttpRequestExecution;
import org.springframework.http.client.ClientHttpRequestInterceptor;
import org.springframework.http.client.ClientHttpResponse;
import org.springframework.http.client.RestTemplate;

import java.io.IOException;

/**
 * API安全最小可运行示例
 * 
 * @since Spring Boot 3.2.5
 * @see <a href="https://spring.io/projects/spring-boot">Spring Boot</a>
 */
public class APISecurityExample {
    private static final Logger logger = LoggerFactory.getLogger(APISecurityExample.class);

    /**
     * 添加API Key认证
     * 
     * @param restTemplate RestTemplate
     * @param apiKey API Key
     */
    public void addAPIKeyAuth(RestTemplate restTemplate, String apiKey) {
        logger.info("Adding API Key authentication");

        // 添加请求拦截器
        restTemplate.getInterceptors().add(new ClientHttpRequestInterceptor() {
            @Override
            public org.springframework.http.HttpResponse execute(
                    HttpRequest request, byte[] body, ClientHttpRequestExecution execution) {
                request.getHeaders().add(HttpHeaders.AUTHORIZATION, "Bearer " + apiKey);
                return execution.execute(request, body);
            }
        });
    }

    /**
     * 主方法，运行API安全最小示例
     */
    public static void main(String[] args) {
        logger.info("API Security Example");

        // 从环境变量读取API Key（生产环境）
        String apiKey = System.getenv("OPENAI_API_KEY");
        if (apiKey == null || apiKey.isEmpty()) {
            logger.warn("OPENAI_API_KEY not found in environment variables");
        }

        RestTemplate restTemplate = new RestTemplate();
        APISecurityExample example = new APISecurityExample();
        example.addAPIKeyAuth(restTemplate, apiKey);
    }
}
```
