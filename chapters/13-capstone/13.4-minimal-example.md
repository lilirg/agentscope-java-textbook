        - name: agentId

            type: string

          description: 请求参数错误

          description: Agent不存在
        '500':
          description: 系统错误

    post:
      summary: 处理请求
      description: 通过Agent处理用户请求
      requestBody:
        required: true
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/ProcessRequest'
      responses:
        '200':
          description: 成功
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ProcessResponse'
        '400':
          description: 请求参数错误
        '500':
          description: 系统错误

components:
  schemas:
    AgentStatus:
      type: object
      properties:
        agentId:
          type: string
          example: "agent-001"
        status:
          type: string
          enum: [RUNNING, STOPPED, ERROR]
          example: "RUNNING"
        cpuUsage:
          type: number
          example: 25.5
        memoryUsage:
          type: number
          example: 512.0
        lastHeartbeat:
          type: string
          format: date-time
          example: "2024-04-15T10:30:00Z"

      type: object
      required: [agentId, query]
      properties:
        agentId:
          type: string
          example: "agent-001"

          type: string

          type: object

              type: string
              example: "user-001"

### 13.4.4 系统实现

**src/main/java/com/agentscope/capstone/api/AgentController.java**：
```java
package com.agentscope.capstone.api;

import com.agentscope.capstone.agent.AgentService;
import com.agentscope.capstone.agent.model.AgentStatus;
import com.agentscope.capstone.agent.model.ProcessRequest;
import com.agentscope.capstone.agent.model.ProcessResponse;
import io.swagger.v3.oas.annotations.Operation;
import io.swagger.v3.oas.annotations.tags.Tag;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

/**
 * Agent控制器
 */
@RestController
@RequestMapping("/api/agent")
@Tag(name = "Agent", description = "Agent管理接口")
public class AgentController {
    
    @Autowired
    private AgentService agentService;
    
    /**
     * 查询Agent状态
     */
    @GetMapping("/status")
    @Operation(summary = "查询Agent状态", description = "根据Agent ID查询Agent状态")
    public ResponseEntity<AgentStatus> getStatus(
            @RequestParam String agentId) {
        AgentStatus status = agentService.getStatus(agentId);
        return ResponseEntity.ok(status);
    }
    
    /**
     * 处理请求
     */
    @PostMapping("/process")
    @Operation(summary = "处理请求", description = "通过Agent处理用户请求")
    public ResponseEntity<ProcessResponse> process(
            @RequestBody ProcessRequest request) {
        ProcessResponse response = agentService.process(request);
        return ResponseEntity.ok(response);
    }
}
```
