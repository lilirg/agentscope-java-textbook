### 9.4.3 核心代码

#### 1. API Key安全（APIConfig.java）

```java
package com.textbook.chapter9;

import io.github.cdimascia.dotenv.Dotenv;

/**
 * API Key配置类
 * @since v9.4.0
 */
public class APIConfig {
    private final String apiKey;
    private final String backupApiKey;
    private final long apiKeyCreatedAt;
    private final boolean isDevEnvironment;

    public APIConfig() {
        this.isDevEnvironment = System.getenv("ENV") != null && 
                                System.getenv("ENV").equals("dev");

        // 1. 从环境变量或.env读取API Key
        Dotenv dotenv = Dotenv.configure().ignoreIfMissing().load();
        
        if (isDevEnvironment) {
            // 开发环境：从环境变量读取
            this.apiKey = System.getenv("API_KEY");
            this.backupApiKey = System.getenv("BACKUP_API_KEY");
            this.apiKeyCreatedAt = System.currentTimeMillis();
        } else {
            // 生产环境：真实Key
            this.apiKey = dotenv.get("API_KEY");
            this.backupApiKey = dotenv.get("BACKUP_API_KEY");
            this.apiKeyCreatedAt = Long.parseLong(dotenv.get("API_KEY_CREATED_AT"));
        }
    }

    public String getApiKey() {
        // 1. 检查密钥有效期（30天）
        long now = System.currentTimeMillis();
        long keyAgeDays = (now - apiKeyCreatedAt) / (1000 * 60 * 60 * 24);

        if (keyAgeDays > 30) {
            // 密钥过期，使用备用密钥
            return backupApiKey;
        }

        return apiKey;
    }

    public boolean isDevEnvironment() {
        return isDevEnvironment;
    }
}
```

#### 2. 加密存储（AESCrypto.java）

```java
package com.textbook.chapter9;

import javax.crypto.Cipher;
import javax.crypto.spec.SecretKeySpec;
import java.util.Base64;

/**
 * AES加密工具类
 * @since v9.4.0
 */
public class AESCrypto {
    private static final String ALGORITHM = "AES";
    private static final String TRANSFORMATION = "AES";

    public static String encrypt(String data, String key) throws Exception {
        SecretKeySpec secretKey = new SecretKeySpec(key.getBytes(), ALGORITHM);
        Cipher cipher = Cipher.getInstance(TRANSFORMATION);
        cipher.init(Cipher.ENCRYPT_MODE, secretKey);
        byte[] encryptedData = cipher.doFinal(data.getBytes());
        return Base64.getEncoder().encodeToString(encryptedData);
    }

    public static String decrypt(String encryptedData, String key) throws Exception {
        SecretKeySpec secretKey = new SecretKeySpec(key.getBytes(), ALGORITHM);
        Cipher cipher = Cipher.getInstance(TRANSFORMATION);
        cipher.init(Cipher.DECRYPT_MODE, secretKey);
        byte[] decodedData = Base64.getDecoder().decode(encryptedData);
        byte[] decryptedData = cipher.doFinal(decodedData);
        return new String(decryptedData);
    }
}
```

#### 3. 权限控制（RBAC.java + Role.java）

```java
package com.textbook.chapter9;

import java.util.Arrays;
import java.util.List;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

/**
 * RBAC权限管理类
 * @since v9.4.0
 */
public class RBAC {
    private final Map<String, Role> userRoles = new ConcurrentHashMap<>();

    public void assignRole(String userId, Role role) {
        userRoles.put(userId, role);
    }

    public boolean hasPermission(String userId, Permission permission) {
        Role role = userRoles.get(userId);
        if (role == null) {
            return false;
        }
        return role.getPermissions().contains(permission);
    }
}
```

#### 4. ABAC权限控制（ABAC.java）

```java
package com.textbook.chapter9;

import java.time.DayOfWeek;
import java.time.LocalDateTime;

/**
 * ABAC权限管理类（基于属性的访问控制）
 * @since v9.4.0
 */
public class ABAC {
    
    public boolean checkAccess(String userId, String resource, String action) {
        // 1. 获取用户属性
        UserAttributes attributes = getUserAttributes(userId);
        
        // 2. 判断权限
        // 限制条件：
        // - 部门必须是IT或HR
        // - 时间必须是工作日（周一至周五）
        // - 地理位置必须是公司内网
        return (attributes.getDepartment().equals("IT") || 
                attributes.getDepartment().equals("HR")) &&
               isWorkday() &&
               isInternalNetwork(attributes.getLocation());
    }

    private UserAttributes getUserAttributes(String userId) {
        // 从数据库/LDAP获取用户属性
        return new UserAttributes(userId, "IT", "Beijing", LocalDateTime.now());
    }

    private boolean isWorkday() {
        DayOfWeek day = LocalDateTime.now().getDayOfWeek();
        return day != DayOfWeek.SATURDAY && day != DayOfWeek.SUNDAY;
    }

    private boolean isInternalNetwork(String location) {
        // 检查IP地址是否在内网范围内
        return location.equals("Beijing") || location.equals("Shanghai");
    }
}
```

```java
package com.textbook.chapter9;

import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.ConcurrentHashMapKeySet;

/**
 * 安全监控服务类
 * @since v9.4.0
 */
public class SecurityMonitor {
    private final Map<String, Integer> apiCallCount = new ConcurrentHashMap<>();
    private final Set<String> suspiciousUsers = ConcurrentHashMapKeySet.newKeySet();

    public void monitorAPICall(String userId) {
        // 1. 统计API调用次数
        apiCallCount.merge(userId, 1, Integer::sum);

        // 2. 检测异常（1分钟超过100次）
        if (apiCallCount.get(userId) > 100) {
            logger.error("API调用异常: userId={}, count={}", userId, apiCallCount.get(userId));
            suspiciousUsers.add(userId);
        }
    }

    public boolean isSuspicious(String userId) {
        return suspiciousUsers.contains(userId);
    }
}
```

#### 6. 异常处理（SecurityExceptionHandler.java）

```java
package com.textbook.chapter9;

import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.RestControllerAdvice;

/**
 * 安全异常处理器
 * @since v9.4.0
 */
@RestControllerAdvice
public class SecurityExceptionHandler {
    
    @ExceptionHandler(ApiKeyLeakException.class)
    public ResponseEntity<ErrorResponse> handleApiKeyLeak(ApiKeyLeakException e) {
        // 1. 记录日志
        log.error("API Key泄露风险: {}", e.getMessage());
        
        // 2. 发送告警（钉钉机器人）
        DingTalkNotifier.notify("API Key泄露风险", e.getMessage());
        
        // 3. 返回错误响应
        ErrorResponse error = new ErrorResponse("SECURITY_ERROR", "API Key泄露风险");
        return ResponseEntity.status(HttpStatus.FORBIDDEN).body(error);
    }
}
```

#### 7. API控制器（Controller）

```java
package com.textbook.chapter9;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

/**
 * API控制器
 * @since v9.4.0
 */
@RestController
@RequestMapping("/api")
public class APIController {
    
    @Autowired
    private APIConfig apiConfig;
    
    @Autowired
    private RBAC rbac;
    
    @Autowired
    private ABAC abac;
    
    @Autowired
    private AuditService auditService;
    
    @Autowired
    private SecurityMonitor securityMonitor;

    @GetMapping("/query")
    public ResponseEntity<String> query(@RequestHeader("X-User-ID") String userId) {
        // 1. API Key验证
        if (apiConfig.isDevEnvironment()) {
            auditService.logAPICall(userId, "/api/query", "dev environment");
        } else {
            // 生产环境：检查API Key
            String apiKey = apiConfig.getApiKey();
            if (apiKey == null || apiKey.isEmpty()) {
                return ResponseEntity.status(HttpStatus.UNAUTHORIZED).body("API Key无效");
            }
        }

        // 2. 权限检查（RBAC + ABAC）
        if (!rbac.hasPermission(userId, Permission.QUERY)) {
            return ResponseEntity.status(HttpStatus.FORBIDDEN).body("无查询权限");
        }

        if (!abac.checkAccess(userId, "query", "GET")) {
            return ResponseEntity.status(HttpStatus.FORBIDDEN).body("不满足ABAC权限条件");
        }

        // 3. 安全审计
        auditService.logAPICall(userId, "/api/query", "");

        // 4. 监控
        securityMonitor.monitorAPICall(userId);

        // 5. 执行查询
        return ResponseEntity.ok("查询成功");
    }

    @DeleteMapping("/delete")
    public ResponseEntity<String> delete(@RequestHeader("X-User-ID") String userId) {
        // 1. 权限检查
        if (!rbac.hasPermission(userId, Permission.DELETE)) {
            return ResponseEntity.status(HttpStatus.FORBIDDEN).body("无删除权限");
        }

        // 2. ABAC检查
        if (!abac.checkAccess(userId, "delete", "DELETE")) {
            return ResponseEntity.status(HttpStatus.FORBIDDEN).body("不满足ABAC权限条件");
        }

        // 3. 执行删除
        return ResponseEntity.ok("删除成功");
    }
}
```
